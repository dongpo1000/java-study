    1、使用 java.util.Date 的常用方法

    # 1、使用 java.util.Date 的常用方法

    package com.atguigu.time8;

import java.util.Date;

/**
 * Description: DateDemo1
 *
 * @author dongpo: Talk is cheap. Show me the code.
 * @last 2023-06-10
 */
public class DateDemo1 {
	public static void main(String[] args) {
		// 生成一个代表当前日期时间的Date对象。该构造器在底层调用了System.currentTimeMillis()获得long整数作为日期参数。
		Date d1 = new Date();
		System.out.println("d1 = " + d1);
		// 返回该时间对应的long型整数，即与GMT 1970-01-01 00:00:00之间的时间差，以毫秒作为单位。
		long t1 = d1.getTime();
		System.out.println("t1 = " + t1);

		System.out.println("=======================");

		// 根据指定的long型整数来生成一个Date对象。该构造器的参数表示创建的Date对象和GMT 1970年1月1日00:00:00之间的时间差，以毫秒作为计时单位。
		Date d2 = new Date(t1 + 100);
		System.out.println("d2 = " + d2);
		long t2 = d2.getTime();
		System.out.println("t2 = " + t2);

		System.out.println("=======================");

		Date d3 = new Date();
		// 设置该Date对象的时间。
		d3.setTime(t1 - 100);
		System.out.println("d3 = " + d3);
		long t3 = d3.getTime();
		System.out.println("t3 = " + t3);

		System.out.println("=======================");

		// 比较时间先后，查看源码可知：大于结果为1，等于结果为0，小于结果为-1
		System.out.println("d1.compareTo(d2) = " + d1.compareTo(d2));
		System.out.println("d1.compareTo(d3) = " + d1.compareTo(d3));
		// 判断时间是否在指定时间之前
		System.out.println("d1.before(d2) = " + d1.before(d2));
		// 判断时间是否在指定时间之后
		System.out.println("d1.after(d2) = " + d1.after(d2));
		System.out.println("d1.before(d3) = " + d1.before(d3));
		System.out.println("d1.after(d3) = " + d1.after(d3));

		/*
			d1 = Sat Jun 10 09:48:05 CST 2023
			t1 = 1686361685608
			=======================
			d2 = Sat Jun 10 09:48:05 CST 2023
			t2 = 1686361685708
			=======================
			d3 = Sat Jun 10 09:48:05 CST 2023
			t3 = 1686361685508
			=======================
			d1.compareTo(d2) = -1
			d1.compareTo(d3) = 1
			d1.before(d2) = true
			d1.after(d2) = false
			d1.before(d3) = false
			d1.after(d3) = true
		 */
	}
}

    # 2、使用 java.time.LocalDateTime 获取当前的年月日时分秒

    package com.atguigu.time8;

import java.time.LocalDateTime;

/**
 * Description: LocalDateTimeDemo1
 *
 * @author dongpo: Talk is cheap. Show me the code.
 * @last 2023-06-10
 */
public class LocalDateTimeDemo1 {
	public static void main(String[] args) {
		LocalDateTime now = LocalDateTime.now();
		System.out.println("now = " + now);

		System.out.println("=================");

		int year = now.getYear();
		System.out.println("year = " + year);
		int monthValue = now.getMonthValue();
		System.out.println("monthValue = " + monthValue);
		int dayOfMonth = now.getDayOfMonth();
		System.out.println("dayOfMonth = " + dayOfMonth);

		System.out.println("=================");

		int hour = now.getHour();
		System.out.println("hour = " + hour);
		int minute = now.getMinute();
		System.out.println("minute = " + minute);
		int second = now.getSecond();
		System.out.println("second = " + second);

		/*
			now = 2023-06-10T10:03:33.377
			=================
			year = 2023
			monthValue = 6
			dayOfMonth = 10
			=================
			hour = 10
			minute = 3
			second = 33
		 */
	}
}

    # 3、 java.util.Date 和 java.time.LocalDateTime 互转

    package com.atguigu.time8;

import java.time.Instant;
import java.time.LocalDateTime;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.util.Date;

/**
 * Description: DateAndLocalDateTimeTransferDemo1
 *
 * @author dongpo: Talk is cheap. Show me the code.
 * @since 2023-06-10
 */
public class DateAndLocalDateTimeTransferDemo1 {
	public static void main(String[] args) {
		// java.util.Date 转 java.time.LocalDateTime
		Date date = new Date();
		System.out.println("date = " + date);
		Instant instant = date.toInstant();
		System.out.println("instant = " + instant);
		ZoneId zoneId = ZoneId.systemDefault();
		System.out.println("zoneId = " + zoneId);
		// 方法一
		LocalDateTime localDateTime = instant.atZone(zoneId).toLocalDateTime();
		System.out.println("localDateTime = " + localDateTime);
		// 方法二
		LocalDateTime localDateTime1 = LocalDateTime.ofInstant(instant, zoneId);
		System.out.println("localDateTime1 = " + localDateTime1);

		System.out.println("===========================");

		// java.time.LocalDateTime 转 java.util.Date
		ZonedDateTime zonedDateTime = localDateTime.atZone(zoneId);
		System.out.println("zonedDateTime = " + zonedDateTime);
		Instant instant1 = zonedDateTime.toInstant();
		Date from = Date.from(instant1);
		System.out.println("from = " + from);

		System.out.println("===========================");

		// 比较
		System.out.println("date.compareTo(from) = " + date.compareTo(from));
		System.out.println("date.equals(from) = " + date.equals(from));

		/*
			date = Sat Jun 10 10:18:50 CST 2023
			instant = 2023-06-10T02:18:50.515Z
			zoneId = Asia/Shanghai
			localDateTime = 2023-06-10T10:18:50.515
			localDateTime1 = 2023-06-10T10:18:50.515
			===========================
			zonedDateTime = 2023-06-10T10:18:50.515+08:00[Asia/Shanghai]
			from = Sat Jun 10 10:18:50 CST 2023
			===========================
			date.compareTo(from) = 0
			date.equals(from) = true
		*/
	}
}
